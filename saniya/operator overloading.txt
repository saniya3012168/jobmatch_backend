#include<iostream>
using namespace std;
class complex{
    private:
    int real,imag;
    public:
    complex(){
        real=0;
        imag=0;
    }
    complex(int r,int i){
        real=r;
        imag=i;
    }
    void add(){
        cout<<"Addition=";
        cout<<real<<"+"<<imag<<"i"<<endl;
    }
    void mul(){
        cout<<"multiplication=";
        cout<<real<<"+"<<imag<<"i"<<endl;
    }
 complex operator +(complex c){    //operator overloading
        complex temp;
        temp.real=real+c.real;
        temp.imag=imag+c.imag;
        return temp;
    }
     complex operator *(complex c){    //operator overloading
        complex temp;
        temp.real=real*c.real;
        temp.imag=imag*c.imag;
        return temp;
    }
};
int main(){
    complex c1(5,6);
    complex c2(7,2);
    complex c3;
    complex c4;
    c4=c1*c2;
    c3=c1+c2;
    c3.add();
    c4.mul();
    
    return 0;
}

OUTPUT:-
Addition=12+8i
multiplication=35+12i
