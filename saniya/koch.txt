void Koch(int x1,int y1,int x2,int y2,int it){
float ang=60*M_PI/180;
int x3=(2*x1+x2)/3;
int y3=(2*y1+y2)/3;
int x4=(x1+2*x2)/3;
int y4=(y1+2*y2)/3;
int x=x1+(x4-x3)*cos(ang)+(y4-y3)*sin(ang);
int y=y1-(x4-x3)*sin(ang)+(y4-y3)*cos(ang);

if(it>0){
Koch(x1,y1,x3,y3,it-1);
Koch(x3,y3,x,y,it-1);
Koch(x,y,x4,y4,it-1);
Koch(x4,y4,x2,y2,it-1);
}
else{
line(x1,y1,x3,y3);
line(x3,y3,x,y);
line(x,y,x4,y4);
line(x4,y4,x2,y2);
}
}
